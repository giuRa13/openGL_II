#version 460 core

in vec3 fragmentColor;
in vec2 fragmentTexCoord;

out vec4 screenColor;

uniform sampler2D material;  //0
uniform sampler2D mask;      //1

void main()
{
    vec3 baseColor = texture(material, fragmentTexCoord).rgb;
    float alpha = texture(mask, fragmentTexCoord).r;

    // screenColor = vec4(fragmentColor, 1.0);              //just colored screen
    // screenColor = texture(material, fragmentTexCoord);   //just image
    screenColor = vec4(baseColor, alpha);                   //image with mask
}
